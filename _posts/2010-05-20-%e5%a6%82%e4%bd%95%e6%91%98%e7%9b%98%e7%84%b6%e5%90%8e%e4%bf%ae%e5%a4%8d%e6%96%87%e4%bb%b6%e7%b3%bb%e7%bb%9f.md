昨天跟光谱学了一招，记记。

<pre class="brush: bash">#! /bin/sh

/data1/apache2/bin/apachectl stop

fuser -k -v -m /data1

umount /data1

ln -s /data0/data1/apache2 /data1/apache2

ln -s /data0/data1/monitor /data1/monitor

/data1/apache2/bin/apachectl start

exit 0
</pre>
